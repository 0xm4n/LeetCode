class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if (!head || !head->next) return head;
        ListNode dummy(-101);
        dummy.next = head;
        ListNode* pre = &dummy;
        while (pre->next) {
            ListNode* cur = pre->next;
            while (cur->next && cur->next->val == cur->val) {
                cur = cur->next;
            }
            if (pre->next == cur) {
                pre = pre->next;
            } else {
                pre->next = cur->next;
            }
        }
        return dummy.next;
        
    }
};
