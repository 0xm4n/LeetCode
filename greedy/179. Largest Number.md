class Solution {
public:
    string largestNumber(vector<int>& nums) {
        vector<string> v;
        for (auto num : nums)
            v.push_back(to_string(num));
        sort(v.begin(), v.end(), [](string &s1, string &s2) {
            return s1 + s2 > s2 + s1;
        });
        string res;
        for (auto s : v)
            res += s;
        while (res[0] == '0' && res.size() > 1)
            res.erase(0, 1);
        return res;
    }
};
